

#call variants with RNA data
varCall.mutect:
cmd	varCall
callertag	mutect
caller	muTect-1.1.4/muTect-1.1.4.jar
programpath	/data/NYGC/Software/muTect/
inputpath	/nethome/bjchen/Projects/Simon/RNA/
outputpath	/nethome/bjchen/Projects/Simon/RNA/varcall/

prefix	mutect
--baqGapOpenPenalty	40
--num_threads	2
--reference_sequence	/nethome/bjchen/Projects/Simon/h37_hg19_chrOnly.fa
-Djava.io.tmpdir	=/scratch/
--analysis_type	MuTect
--dbsnp	/data/NYGC/Resources/muTect/dbsnp_132_b37.leftAligned.vcf
--cosmic	/data/NYGC/Resources/muTect/hg19_cosmic_v54_120711.vcf
--intervals	/data/NYGC/Resources/muTect/GRCh37_64.chrom.intervals
bam	=sample
normalsample	Sample_132540-10N_Aligned.out.WithReadGroup.sorted..mdup.bam; Sample_138381-4N_Aligned.out.WithReadGroup.sorted.mdup.bam

overwrite	True
mem	24G
